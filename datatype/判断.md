##1.typeof

```
// object
typeof {a:1}        ------------> 'object'
typeof []           ------------> 'object'
typeof [1,2,3]      ------------> 'object'
typeof null         ------------> 'object'
typeof new Date()   ------------> 'object'

//string
typeof '1'          ------------> 'string'
typeof ''           ------------> 'string'
typeof (typeof(1))  ------------> 'string'

//number
typeof 1            ------------> 'number'
typeof NaN          ------------> 'number'

//boolean
typeof true         ------------> 'boolean'
typeof false        ------------> 'boolean'

// undefined
typeof undefined    ------------> 'undefined'
typeof undeclaredVariable ----> 'undefined'

//function
typeof Object       ------------> 'function'
typeof function(){} ------------> 'function'
typeof alert        ------------> 'function'

```
